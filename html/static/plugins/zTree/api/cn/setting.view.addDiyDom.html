Function(treeId, treeNode)setting.view.addDiyDom

Overview[ Dependency jquery.ztree.core Core js ]

用于在Nodes上固定ShowUserCustom Controls

1. 大Data量的Nodes加载请Attention：在 addDiyDom 中针对每个Nodes Find DOM Object并且Add新 DOM Controls，肯定It affectsInitialize性能；If不是必须Use，Recommendations不Use此Functions

2. 属于Advanced应用，Use时请确保对 zTree More。

Default value：null

Function Parameters说明

treeIdString

Correspond zTree 的 treeId，便于User操控

treeNodeJSON

需要ShowCustom Controls的Nodes JSON DataObject

setting & function Example

1. SettingsNodes后面Show一个button

var setting = {
	view: {
		addDiyDom: addDiyDom
	}
};
function addDiyDom(treeId, treeNode) {
	var aObj = $("#" + treeNode.tId + "_a");
	if ($("#diyBtn_"+treeNode.id).length>0) return;
	var editStr = "<span id='diyBtn_space_" +treeNode.id+ "' > </span>"
		+ "<button type='button' class='diyBtn1' id='diyBtn_" + treeNode.id
		+ "' title='"+treeNode.name+"' onfocus='this.blur();'></button>";
	aObj.append(editStr);
	var btn = $("#diyBtn_"+treeNode.id);
	if (btn) btn.bind("click", function(){alert("diy Button for " + treeNode.name);});
};
