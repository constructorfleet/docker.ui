Function(treeId, treeNode)setting.view.addHoverDom

Overview[ Dependency jquery.ztree.exedit Extension js ]

用于当MouseMove到Nodes上时，ShowUserCustom Controls，ShowHideStatus同 zTree Internal editing、Deletebutton

Please be with setting.view.removeHoverDom 同时Use；属于Advanced应用，Use时请确保对 zTree More。

Default value：null

Function Parameters说明

treeIdString

Correspond zTree 的 treeId，便于User操控

treeNodeJSON

需要ShowCustom Controls的Nodes JSON DataObject

setting & function Example

1. SettingsMouse移到Nodes上，在后面Show一个button

var setting = {
	view: {
		addHoverDom: addHoverDom,
		removeHoverDom: removeHoverDom,
		......
	}
};
function addHoverDom(treeId, treeNode) {
	var aObj = $("#" + treeNode.tId + "_a");
	if ($("#diyBtn_"+treeNode.id).length>0) return;
	var editStr = "<span id='diyBtn_space_" +treeNode.id+ "' > </span>"
		+ "<button type='button' class='diyBtn1' id='diyBtn_" + treeNode.id
		+ "' title='"+treeNode.name+"' onfocus='this.blur();'></button>";
	aObj.append(editStr);
	var btn = $("#diyBtn_"+treeNode.id);
	if (btn) btn.bind("click", function(){alert("diy Button for " + treeNode.name);});
};
function removeHoverDom(treeId, treeNode) {
	$("#diyBtn_"+treeNode.id).unbind().remove();
	$("#diyBtn_space_" +treeNode.id).unbind().remove();
};
